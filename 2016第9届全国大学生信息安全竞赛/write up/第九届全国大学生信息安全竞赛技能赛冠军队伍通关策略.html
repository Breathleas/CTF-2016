def mcdecode(inp):
    data=bin(int(inp,16))
    s=data[2:]
    if len(s)%2 !=0:
        s='0'+s
    res=0
    count = 0
    ans=""
    for i in range(0,len(s),2):
        if s[i:i+2]=="10":
            res = (res << 1) | 0
            count += 1
        if s[i:i+2]=="01":
            res = (res << 1) | 1
            count += 1
        if count == 8:
            ans += chr(res)
            count = 0
            res = 0
    a = ''
    for i in ans:
        tmp = 0
        t = ord(i)
        for j in range(8):
            tmp = tmp << 1
            tmp |= t & 0x01
            t = t >> 1
        a = a + chr(tmp)
    print a.encode('hex')

1. 先用**.**让指针往上移，到一定位置泄露出栈地址。
